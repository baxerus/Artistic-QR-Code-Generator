const PERFECT_RESULT_THRESHOLD = 5;  // OPT-01: Auto-stop when this many 0-error results found

/**
 * Check if auto-stop condition is met (OPT-01)
 * @param {Array} topResults - Current merged top results
 * @returns {boolean} True if 5+ results have 0 pixel errors AND are decodable
 */
function checkAutoStopCondition(topResults) {
  if (!topResults || topResults.length < PERFECT_RESULT_THRESHOLD) {
    return false;
  }
  const perfectCount = topResults.filter(r => r.decodable && r.pixelDiff === 0).length;
  return perfectCount >= PERFECT_RESULT_THRESHOLD;
}

function handleSearchComplete(data) {
  const optimizeBtn = document.getElementById("optimize-btn");
  const searchProgress = document.getElementById("search-progress");
  const searchSummary = document.getElementById("search-summary");

  // Clear progress interval
  if (progressInterval) {
    clearInterval(progressInterval);
    progressInterval = null;
  }

  // Update state
  searchState = "completed";
  optimizeBtn.textContent = "Run again";
  searchProgress.style.display = "none";
  const topResultsEl = document.getElementById("top-results");
  if (topResultsEl) topResultsEl.classList.remove("results-disabled");

  // Store results for accumulation
  currentTopResults = data.topResults;

  // Accumulate stats across runs
  totalAttempts += data.attempts;
  totalDecodableCount += data.decodableCount || 0;

  // OPT-01: Show auto-stop or normal completion message
  if (data.autoStopped) {
    searchSummary.textContent = `Auto-stopped - 5 perfect results found! (${totalAttempts.toLocaleString()} attempts)`;
  } else {
    searchSummary.textContent = `Done - ${totalAttempts.toLocaleString()} attempts, ${totalDecodableCount} decodable found`;
  }
  searchSummary.style.display = "flex";

  // Final update to top results
  if (data.topResults && data.topResults.length > 0) {
    updateTopResults(data.topResults);
  }
}
