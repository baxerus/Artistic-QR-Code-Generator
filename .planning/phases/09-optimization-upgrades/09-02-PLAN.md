/**
 * Create message handler for a specific worker (OPT-02)
 * @param {number} workerId - Worker index for tracking
 * @returns {function} Message handler function
 */
function createWorkerMessageHandler(workerId) {
  return function(e) {
    if (e.data.type === 'PROGRESS') {
      // Update this worker's buffers
      workerResultBuffers[workerId] = e.data.topResults || [];
      workerAttemptCounts[workerId] = e.data.attempts || 0;
      workerDecodableCounts[workerId] = e.data.decodableCount || 0;
      
      // Merge results from all workers
      currentTopResults = mergeWorkerResults(workerResultBuffers);
      
      // Calculate total stats across all workers
      const totalAttemptsSoFar = workerAttemptCounts.reduce((a, b) => a + b, 0);
      const totalDecodableSoFar = workerDecodableCounts.reduce((a, b) => a + b, 0);
      
      // Update UI with merged data
      updateSearchStats(totalAttemptsSoFar, e.data.elapsed);
      if (currentTopResults.length > 0) {
        updateTopResults(currentTopResults);
      }
      
      // OPT-01: Check auto-stop condition after merge
      if (checkAutoStopCondition(currentTopResults)) {
        console.log('Auto-stop triggered: 5 perfect results found');
        stopAllWorkers();
        handleSearchComplete({
          attempts: totalAttemptsSoFar,
          decodableCount: totalDecodableSoFar,
          topResults: currentTopResults,
          autoStopped: true
        });
      }
      
    } else if (e.data.type === 'COMPLETE') {
      completedWorkerCount++;
      workerResultBuffers[workerId] = e.data.topResults || [];
      workerAttemptCounts[workerId] = e.data.attempts || 0;
      workerDecodableCounts[workerId] = e.data.decodableCount || 0;
      
      console.log(`Worker ${workerId} complete (${completedWorkerCount}/${WORKER_COUNT})`);
      
      // All workers done?
      if (completedWorkerCount >= WORKER_COUNT) {
        finalizeParallelOptimization();
      }
    }
  };
}

/**
 * Merge results from all workers into single top-5 (OPT-02)
 * Deduplicates by hash, sorts by decodable then pixelDiff
 * @param {Array<Array>} buffers - Per-worker result arrays
 * @returns {Array} Merged and sorted top-5 results
 */
function mergeWorkerResults(buffers) {
  // Flatten all worker results
  const combined = buffers.flat();
  
  // Deduplicate by hash (same hash = same QR configuration)
  const seen = new Set();
  const unique = combined.filter(r => {
    if (!r || !r.hash) return false;
    if (seen.has(r.hash)) return false;
    seen.add(r.hash);
    return true;
  });
  
  // Sort: decodable first, then by pixelDiff ascending
  unique.sort((a, b) => {
    if (a.decodable !== b.decodable) return b.decodable ? 1 : -1;
    return a.pixelDiff - b.pixelDiff;
  });
  
  return unique.slice(0, 5);
}

  // Create blob URL once, reuse for all workers
  const blob = new Blob([workerCode], { type: 'application/javascript' });
  workerBlobUrl = URL.createObjectURL(blob);
  
  // Reset state
  optimizationWorkers = [];
  workerResultBuffers = new Array(WORKER_COUNT).fill([]);
  workerAttemptCounts = new Array(WORKER_COUNT).fill(0);
  workerDecodableCounts = new Array(WORKER_COUNT).fill(0);
  completedWorkerCount = 0;
  
  // Create workers
  for (let i = 0; i < WORKER_COUNT; i++) {
    const worker = new Worker(workerBlobUrl);
    worker.workerId = i;
    worker.onmessage = createWorkerMessageHandler(i);
    worker.onerror = (err) => console.error(`Worker ${i} error:`, err);
    optimizationWorkers.push(worker);
  }
  
  console.log(`Created worker pool with ${WORKER_COUNT} workers`);
  return optimizationWorkers;
}

/**
 * Terminate all workers and clean up (OPT-02)
 */
function terminateWorkerPool() {
  optimizationWorkers.forEach(w => w.terminate());
  optimizationWorkers = [];
  workerResultBuffers = [];
  workerAttemptCounts = [];
  workerDecodableCounts = [];
  completedWorkerCount = 0;
  
  if (workerBlobUrl) {
    URL.revokeObjectURL(workerBlobUrl);
    workerBlobUrl = null;
  }
}

/**
 * Stop all workers (OPT-02)
 */
function stopAllWorkers() {
  optimizationWorkers.forEach(worker => {
    worker.postMessage({ type: 'STOP_OPTIMIZATION' });
  });
}

 * Finalize parallel optimization when all workers complete (OPT-02)

  const totalWorkerAttempts = workerAttemptCounts.reduce((a, b) => a + b, 0);

}

/**
 * Start parallel optimization search (OPT-02)
 * @param {number} timeout - Timeout in milliseconds
 */
function startOptimization(timeout = 10000) {
  if (!currentURL || !currentVersion || !paintGrid) {
    console.error('Cannot start optimization: missing URL, version, or paint grid');
    return;
  }
  
  createWorkerPool();
  
  const message = {
    type: 'START_OPTIMIZATION',
    baseURL: currentURL,
    version: currentVersion,
    paintedPixels: Array.from(paintGrid.cells),
    functionMask: Array.from(createFunctionPatternMask(currentVersion)),
    timeout: timeout,
    existingTopResults: currentTopResults,
  };
  
  console.log(`Starting parallel optimization with ${WORKER_COUNT} workers, timeout ${timeout}ms`);
  
  // Send to all workers simultaneously
  optimizationWorkers.forEach(worker => {
    worker.postMessage(message);
  });
}

/**
 * Stop optimization search (OPT-02)
 */
function stopOptimization() {
  stopAllWorkers();
}

/**
 * Terminate optimization worker pool (OPT-02)

function terminateOptimizationWorker() {
  terminateWorkerPool();
}
