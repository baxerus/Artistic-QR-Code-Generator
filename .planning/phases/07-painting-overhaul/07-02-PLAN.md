    3. Create coordinate transformation function:
       ```javascript
       function getGridCoordinates(canvas, event) {
         const rect = canvas.getBoundingClientRect();
         const scaleX = canvas.width / rect.width;
         const scaleY = canvas.height / rect.height;
         
         const canvasX = (event.clientX - rect.left) * scaleX;
         const canvasY = (event.clientY - rect.top) * scaleY;
         
         const cellSize = canvas.width / paintGrid.size;
         
         // Apply sensitivity tuning
         const rawX = canvasX / cellSize;
         const rawY = canvasY / cellSize;
         const gridX = Math.floor(rawX + (1 - GRID_SENSITIVITY) * 0.5);
         const gridY = Math.floor(rawY + (1 - GRID_SENSITIVITY) * 0.5);
         
         return { x: gridX, y: gridY };
       }
       ```

    4. Create painting logic:
       ```javascript
       function paintAt(event) {
         const qrCanvas = document.querySelector('#qr-canvas-container canvas');
         if (!qrCanvas || !paintGrid) return;
         
         const coords = getGridCoordinates(qrCanvas, event);
         const { x, y } = coords;
         
         // Bounds check
         if (x < 0 || x >= paintGrid.size || y < 0 || y >= paintGrid.size) return;
         
         const cellKey = `${x},${y}`;
         
         // Debounce: don't repaint same cell during drag
         if (lastPaintedCell === cellKey) return;
         
         // Check if protected - transparent to drag, just skip painting
         const functionMask = createFunctionPatternMask(currentVersion);
         if (functionMask[y * paintGrid.size + x]) {
           lastPaintedCell = cellKey; // Mark as visited but don't paint
           return;
         }
         
         // Determine paint value based on button
         const paintValue = getPaintValue(event.button);
         if (paintValue === null) return; // Nothing to paint
         
         // Apply paint
         paintGrid.set(x, y, paintValue);
         lastPaintedCell = cellKey;
         
         // Re-render with overlay
         const qrModel = currentQRModel; // Need to store this globally
         renderQRWithOverlay(qrCanvas, qrModel, paintGrid, functionMask, true);
         
         // Run corruption pipeline to update QR display
         runCorruptionPipeline();
         
         // Reset optimization results
         resetOptimizationResults();
         
         // Check paint state
         checkPaintState();
         
         // Save pattern
         savePattern(currentVersion, paintGrid.cells);
       }
       ```

    5. Create getPaintValue function:
       ```javascript
       function getPaintValue(button) {
         if (button === 0) {
           // Left click: paint selected color
           if (selectedColor === 'black') return 2;
           if (selectedColor === 'white') return 1;
           if (selectedColor === 'unset') return 0;
         } else if (button === 2) {
           // Right click: paint opposite color
           if (selectedColor === 'unset') return null; // Does nothing
           if (selectedColor === 'black') return 1; // Opposite of black is white
           if (selectedColor === 'white') return 2; // Opposite of white is black
         }
         return null;
       }
       ```

    6. Add pointer event listeners to QR canvas container:
       ```javascript
       function setupPaintingEvents() {
         const container = document.getElementById('qr-canvas-container');
         
         container.addEventListener('pointerdown', (e) => {
           if (e.button !== 0 && e.button !== 2) return; // Only left/right mouse
           if (searchState === 'running') return; // Block during optimization
           
           isPainting = true;
           container.setPointerCapture(e.pointerId);
           paintAt(e);
         });
         
         container.addEventListener('pointermove', (e) => {
           if (!isPainting) {
             // Just update cursor for hover feedback
             updateCursor(e);
             return;
           }
           paintAt(e);
         });
         
         container.addEventListener('pointerup', (e) => {
           isPainting = false;
           lastPaintedCell = null;
           container.releasePointerCapture(e.pointerId);
         });
         
         container.addEventListener('pointerleave', () => {
           // Don't stop painting on leave - pointer capture handles this
         });
         
         // Prevent context menu on right-click
         container.addEventListener('contextmenu', (e) => {
           e.preventDefault();
         });
       }
       ```

    7. Create updateCursor function for hover feedback:
       ```javascript
       function updateCursor(event) {
         const container = document.getElementById('qr-canvas-container');
         const canvas = container.querySelector('canvas');
         if (!canvas) return;
         
         const coords = getGridCoordinates(canvas, event);
         const { x, y } = coords;
         
         if (x < 0 || x >= paintGrid.size || y < 0 || y >= paintGrid.size) return;
         
         const functionMask = createFunctionPatternMask(currentVersion);
         const isProtected = functionMask[y * paintGrid.size + x];
         
         if (isProtected) {
           container.classList.add('blocked');
           container.classList.remove('paintable');
         } else {
           container.classList.add('paintable');
           container.classList.remove('blocked');
         }
       }
       ```
