1. Create a new function `renderProtectedBordersOverlay` (place near `renderErrorOverlay` around line 15876):
```javascript
/**
 * Render protected area borders as an overlay (blue dashed)
 * @param {HTMLCanvasElement} canvas - Canvas to render on
 * @param {number} moduleCount - Size of QR code
 * @param {Array<boolean>} functionMask - Function pattern mask
 */
function renderProtectedBordersOverlay(canvas, moduleCount, functionMask) {
  const ctx = canvas.getContext("2d");
  ctx.strokeStyle = PROTECTED_OUTLINE_COLOR;
  ctx.lineWidth = 0.15; // Thin line for module-scale canvas
  ctx.setLineDash([0.5, 0.25]); // Scaled dashes for module resolution
  
  for (let y = 0; y < moduleCount; y++) {
    for (let x = 0; x < moduleCount; x++) {
      const idx = y * moduleCount + x;
      if (!functionMask[idx]) continue;
      
      // Draw border only on outer edges
      // Top
      if (y === 0 || !functionMask[(y - 1) * moduleCount + x]) {
        ctx.beginPath();
        ctx.moveTo(x, y);
        ctx.lineTo(x + 1, y);
        ctx.stroke();
      }
      // Bottom
      if (y === moduleCount - 1 || !functionMask[(y + 1) * moduleCount + x]) {
        ctx.beginPath();
        ctx.moveTo(x, y + 1);
        ctx.lineTo(x + 1, y + 1);
        ctx.stroke();
      }
      // Left
      if (x === 0 || !functionMask[y * moduleCount + (x - 1)]) {
        ctx.beginPath();
        ctx.moveTo(x, y);
        ctx.lineTo(x, y + 1);
        ctx.stroke();
      }
      // Right
      if (x === moduleCount - 1 || !functionMask[y * moduleCount + (x + 1)]) {
        ctx.beginPath();
        ctx.moveTo(x + 1, y);
        ctx.lineTo(x + 1, y + 1);
        ctx.stroke();
      }
    }
  }
  ctx.setLineDash([]);
}
```

2. Call this function after `renderErrorOverlay` in the eager rendering (Task 2):
```javascript

```

3. Since both overlays render to the same canvas and are shown via CSS on hover, the blue dashed borders will appear alongside the error visualization.

**Note:** The overlay canvas uses module-scale coordinates (1 module = 1 pixel), so line widths and dash patterns need to be scaled down accordingly. CSS `image-rendering: pixelated` handles the upscaling.
